```
gnome-shell --version 2>/dev/null
sudo apt-get update && sudo apt-get install \
  gnome-shell-extension-manager \
  gnome-browser-connector
sudo apt-get update && sudo apt-get install \
  gir1.2-gtop-2.0 \
  gir1.2-nm-1.0 \
  gnome-system-monitor
```
Restart GNOME Shell: `Alt+F2 r`
